# UI Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build the React UI: routing, layout, control bar, two-section form list with custom enter-key focus order, the connections graph view (inline on wide viewports, bottom sheet on narrow), the about page, and the load/save/export wiring.

**Architecture:** React Router gives us `/` and `/about`. A `RootLayout` provides the navigation bar (which collapses to a drawer on narrow viewports). On the main page, a `ControlBar` sits above a two-column main layout: connections graph (hidden on narrow) on the left and the form-list view on the right. The form list iterates `formClassOrder`, rendering a `FormClassTable` per class; cells dispatch into the Zustand store. A `FocusContext` registry implements the special enter-key focus order. The connections view is a thin React Flow integration over data from the connections-graph extractor. Depends on every previous plan.

**Tech Stack:** React 19, React Router 7, React Aria Components, React Flow, CSS modules.

> **Scope note:** This plan covers the major components and the integration glue. Pixel-level styling and visual polish are out of scope; the goal is functional, accessible UI with project-rule conformance.

```typescript
      formClassOrder: state.formClassOrder,

    const url = URL.createObjectURL(blob);
    const a = document.createElement("a");
    a.href = url;
    a.download = "thumbtax-save.json";
    a.click();
    URL.revokeObjectURL(url);
  };

  return <Button onPress={onPress}>Download</Button>;
```

```typescript
import { FileTrigger } from "react-aria-components";

import { deserialize } from "#src/persistence/deserialize";
import { useStore } from "#src/state/store";
import { Button } from "#src/ui/primitives/Button";

export function UploadButton() {
  const loadState = useStore((s) => s.loadState);

  const onSelect = async (fileList: FileList | null) => {
    if (!fileList || fileList.length === 0) return;
    const text = await fileList[0].text();
    const { state } = deserialize(JSON.parse(text));
    loadState(state);
  };

  return (
    <FileTrigger acceptedFileTypes={["application/json"]} onSelect={onSelect}>
      <Button>Upload</Button>
    </FileTrigger>
  );
}
```

```typescript
import { Dialog, DialogTrigger, Heading } from "react-aria-components";

import {
  allFormClasses,
  getFormSpecification,
} from "#src/specifications/service";
import { useStore } from "#src/state/store";
import { Button } from "#src/ui/primitives/Button";
import { Modal } from "#src/ui/primitives/Modal";

type Props = { trigger: React.ReactNode };

export function AddFormPicker({ trigger }: Props) {
  const addFormInstance = useStore((s) => s.addFormInstance);
  const formInstancesByClass = useStore((s) => s.formInstancesByClass);

  return (
    <DialogTrigger>
      {trigger}
      <Modal>
        <Dialog>
          {({ close }) => (
            <>
              <Heading slot="title">Add a form</Heading>
              {allFormClasses().map((cls) => {
                const spec = getFormSpecification(cls);
                const present =
                  formInstancesByClass[cls]?.length ?? 0;
                const atMax =
                  spec.maxInstances !== null && present >= spec.maxInstances;
                return (
                  <Button
                    key={cls}
                    isDisabled={atMax}
                    onPress={() => {
                      if (!atMax) {
                        addFormInstance(cls);
                        close();
                      }
                    }}
                  >
                    {spec.title}
                    {spec.subtitle ? ` — ${spec.subtitle}` : ""}
                  </Button>
                );
              })}
            </>
          )}
        </Dialog>
      </Modal>
    </DialogTrigger>
  );
}
```

```typescript
import type { FormClass } from "#src/common/types/formClass";
import type { TaxFormBox } from "#src/specifications/types/taxFormBox";
import type { TaxFormSpecification } from "#src/specifications/types/taxFormSpecification";
import type { PrimaryState } from "#src/state/types/primaryState";
import type { FocusKey } from "#src/ui/formList/focus/types/focusKey";

type Specs = Map<FormClass, TaxFormSpecification>;

function inputBoxesOnLine(
  spec: TaxFormSpecification,
  lineIndex: string,
): TaxFormBox[] {
  for (const section of spec.sections) {
    for (const line of section.lines) {
      if (line.index === lineIndex) {
        return line.boxes.filter((b) =>
          ["number_input", "checkbox_input", "selection_input", "list_amounts_input"].includes(
            b.value.type,
          ),
        );
      }
    }
  }
  return [];
}

function allInputLines(spec: TaxFormSpecification): string[] {
  const out: string[] = [];
  for (const section of spec.sections) {
    for (const line of section.lines) {
      if (inputBoxesOnLine(spec, line.index).length > 0) out.push(line.index);
    }
  }
  return out;
}

export function computeNextFocusKey(
  current: FocusKey,
  context: { specs: Specs; state: PrimaryState },
): FocusKey | null {
  const { specs, state } = context;
  const spec = specs.get(current.class);
  if (!spec) return null;

  const inputBoxes = inputBoxesOnLine(spec, current.line);
  const colIdx = current.column
    ? inputBoxes.findIndex((b) => b.columnIndex === current.column)
    : -1;
  if (colIdx >= 0 && colIdx < inputBoxes.length - 1) {
    return {
      class: current.class,
      instance: current.instance,
      line: current.line,
      column: inputBoxes[colIdx + 1].columnIndex,
    };
  }

  const lines = allInputLines(spec);
  const lineIdx = lines.indexOf(current.line);
  if (lineIdx >= 0 && lineIdx < lines.length - 1) {
    const nextLine = lines[lineIdx + 1];
    const firstCol = inputBoxesOnLine(spec, nextLine)[0]?.columnIndex;
    return {
      class: current.class,
      instance: current.instance,
      line: nextLine,
      column: firstCol,
    };
  }

  const instances = state.formInstancesByClass[current.class] ?? [];
  const ix = instances.findIndex((i) => i.instanceId === current.instance);
  if (ix >= 0 && ix < instances.length - 1) {
    const nextInstance = instances[ix + 1];
    const firstLine = lines[0];
    if (!firstLine) return null;
    const firstCol = inputBoxesOnLine(spec, firstLine)[0]?.columnIndex;
    return {
      class: current.class,
      instance: nextInstance.instanceId,
      line: firstLine,
      column: firstCol,
    };
  }

  return null;
}
```

```typescript
import { createContext, useContext, useMemo, useRef, type ReactNode } from "react";

import { computeNextFocusKey } from "#src/ui/formList/focus/computeNextFocusKey";
import type { FocusKey } from "#src/ui/formList/focus/types/focusKey";
import type { FocusRegistry } from "#src/ui/formList/focus/types/focusRegistry";
import {
  allFormClasses,
  getFormSpecification,
} from "#src/specifications/service";
import { useStore } from "#src/state/store";

const SPECS = new Map(
  allFormClasses().map((c) => [c, getFormSpecification(c)]),
);

const FocusRegistryContext = createContext<FocusRegistry | null>(null);

function encodeKey(key: FocusKey): string {
  return `${key.class}|${key.instance}|${key.line}|${key.column ?? ""}`;
}

export function FocusProvider({ children }: { children: ReactNode }) {
  const elements = useRef(new Map<string, HTMLElement>());
  const getState = useStore.getState;

  const registry = useMemo<FocusRegistry>(
    () => ({
      register(key, el) {
        const k = encodeKey(key);
        elements.current.set(k, el);
        return () => {
          if (elements.current.get(k) === el) elements.current.delete(k);
        };
      },
      focusNext(from) {
        const next = computeNextFocusKey(from, {
          specs: SPECS,
          state: getState(),
        });
        if (!next) return;
        elements.current.get(encodeKey(next))?.focus();
      },
      focusPrevious() {
        // Symmetric to focusNext; default tab handles reverse navigation.
      },
    }),
    [getState],
  );

  return (
    <FocusRegistryContext.Provider value={registry}>
      {children}
    </FocusRegistryContext.Provider>
  );
}

export function useFocusRegistry(): FocusRegistry {
  const r = useContext(FocusRegistryContext);
  if (!r) throw new Error("FocusProvider missing");
  return r;
}
```

```typescript
import { useEffect, useRef } from "react";

import type { BoxIdentifier } from "#src/common/types/boxIdentifier";
import type { FormClass } from "#src/common/types/formClass";
import type { FormInstanceId } from "#src/common/types/formInstanceId";
import { formatBoxValue } from "#src/exporters/formatBoxValue";
import { getFormSpecification } from "#src/specifications/service";
import { useResolvedBox, useStore } from "#src/state/store";
import { useFocusRegistry } from "#src/ui/formList/focus/FocusContext";
import { Checkbox } from "#src/ui/primitives/Checkbox";
import { NumberField } from "#src/ui/primitives/NumberField";

type Props = {
  formClass: FormClass;
  instanceId: FormInstanceId;
  boxId: BoxIdentifier;
  lineIndex: string;
};

function findBox(formClass: FormClass, boxId: BoxIdentifier) {
  for (const section of getFormSpecification(formClass).sections) {
    for (const line of section.lines) {
      for (const box of line.boxes) {
        if (box.identifier === boxId) return box;
      }
    }
  }
  return undefined;
}

export function FormBoxCell({
  formClass,
  instanceId,
  boxId,
  lineIndex,
}: Props) {
  const box = findBox(formClass, boxId);
  const { input, value } = useResolvedBox(formClass, instanceId, boxId);
  const setBoxInput = useStore((s) => s.setBoxInput);
  const registry = useFocusRegistry();
  const elementRef = useRef<HTMLElement | null>(null);

  useEffect(() => {
    if (!elementRef.current) return;
    return registry.register(
      { class: formClass, instance: instanceId, line: lineIndex, column: box?.columnIndex },
      elementRef.current,
    );
  }, [registry, formClass, instanceId, lineIndex, box?.columnIndex]);

  if (!box) return null;

  const onKeyDown = (event: React.KeyboardEvent) => {
    if (event.key === "Enter") {
      event.preventDefault();
      registry.focusNext({
        class: formClass,
        instance: instanceId,
        line: lineIndex,
        column: box.columnIndex,
      });
    }
  };

  switch (box.value.type) {
    case "number_input":
      return (
        <NumberField
          label={boxId}
          value={
            input && input.type === "number" ? input.value : NaN
          }
          onBlur={(event) => {
            const raw = (event.target as HTMLInputElement).value;
            const parsed = Number.parseFloat(raw);
            const nextValue = Number.isFinite(parsed) ? parsed : 0;
            setBoxInput(formClass, instanceId, boxId, {
              type: "number",
              value: nextValue,
            });
          }}
          onKeyDown={onKeyDown}
          ref={(el) => {
            elementRef.current = (el as HTMLInputElement) ?? null;
          }}
        />
      );
    case "checkbox_input":
      return (
        <Checkbox
          label={boxId}
          isSelected={!!(input && input.type === "number" && input.value !== 0)}
          onChange={(checked: boolean) =>
            setBoxInput(formClass, instanceId, boxId, {
              type: "number",
              value: checked ? 1 : 0,
            })
          }
        />
      );
    case "unused":
      return <span>—</span>;
    case "unsupported":
      return <span>N/A</span>;
    default:
      return <span>{formatBoxValue(value, box.format)}</span>;
  }
}
```

(Note: precise React Aria ref forwarding is library-specific — adapt to API surface. The intent is: this cell registers its DOM element so the focus registry can target it.)
